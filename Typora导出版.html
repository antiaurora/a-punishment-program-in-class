int main(){
    double x1=0,x2=0,x3=0; 
    int count=20; 
    for(int i=0;i<count;i++){ 
        double nx1=x1,nx2=x2,nx3=x3; 
        printf("第%d次:nx1=%lf nx2=%f nx3=%lf\n",i,nx1,nx2,nx3); 
        nx1=(-0.4)*x2-0.2*x3-2.4; 
        nx2=0.25*x1-0.5*x3+5; 
        nx3=(-0.2*x1)+0.3*x2+0.3; 
    x1=nx1; x2=nx2; x3=nx3; 
    }
    return 0;
}
